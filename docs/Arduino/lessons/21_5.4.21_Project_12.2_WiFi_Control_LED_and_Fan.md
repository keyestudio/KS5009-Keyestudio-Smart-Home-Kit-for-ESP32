### 5.4.21 Project 12.2 WiFi-besturing van LED en ventilator


#### **1. Beschrijving**

In dit project leren we hoe we verschillende functies van het slimme huis kunnen realiseren door verschillende strings onder het adres te benaderen. Er is een LCD-scherm dat het IP-adres kan weergeven, wat veel handiger is.


#### **2. Testcode**

⚠️ \ **LET OP:**\  Na het openen van het codebestand moet je de WiFi-naam en wachtwoorden wijzigen waarmee het ESP32 ontwikkelbord verbinding moet maken. Vervang `ChinaNet-2.4G-0DF0` en `ChinaNet@233` door respectievelijk je eigen WiFi-naam en wachtwoord. Je moet dit doen voordat je de code uploadt; anders kan het ESP32 ontwikkelbord geen verbinding maken met het netwerk.

```c
const char* ssid = "ChinaNet-2.4G-0DF0";  // Enter your own WiFi name
const char* password = "ChinaNet@233"; // Enter your own WiFi passwords
```
⚠️ **OPMERKING: Zorg ervoor dat de WiFi-naam en wachtwoorden in de code hetzelfde zijn als het netwerk waarmee je computer, mobiele telefoon/tablet, ESP32 development board en router zijn verbonden. Ze moeten zich binnen hetzelfde lokale netwerk (WiFi) bevinden.**

⚠️ **OPMERKING: De WiFi moet op een 2.4Ghz-frequentie werken; anders kan de ESP32 geen verbinding maken met WiFi.**

```c
#include <Arduino.h>
#include <WiFi.h>
#include <ESPmDNS.h>
#include <WiFiClient.h>

String item = "0";
const char* ssid = "ChinaNet-2.4G-0DF0";
const char* password = "ChinaNet@233";
WiFiServer server(80);

#include <Wire.h>
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C mylcd(0x27,16,2);
//#include <analogWrite.h>
#define fanPin1 19
#define fanPin2 18
#define led_y 12  //Define yellow LED pin as 12

void setup() {
  Serial.begin(115200);
  mylcd.init();
  mylcd.backlight();
  pinMode(led_y, OUTPUT);
  pinMode(fanPin1, OUTPUT);
  pinMode(fanPin2, OUTPUT);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.print("Connected to ");
  Serial.println(ssid);
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
  server.begin();
  Serial.println("TCP server started");
  MDNS.addService("http", "tcp", 80);
  mylcd.setCursor(0, 0);
  mylcd.print("ip:");
  mylcd.setCursor(0, 1);
  mylcd.print(WiFi.localIP());  //LCD displays IP address
}

void loop() {
  WiFiClient client = server.available();
  if (!client) {
      return;
  }
  while(client.connected() && !client.available()){
      delay(1);
  }
  String req = client.readStringUntil('\r');
  int addr_start = req.indexOf(' ');
  int addr_end = req.indexOf(' ', addr_start + 1);
  if (addr_start == -1 || addr_end == -1) {
      Serial.print("Invalid request: ");
      Serial.println(req);
      return;
  }
  req = req.substring(addr_start + 1, addr_end);
  item=req;
  Serial.println(item);
  String s;
  if (req == "/")  //Browser can read the information sent by client.println(s) when accessing the address
  {
      IPAddress ip = WiFi.localIP();
      String ipStr = String(ip[0]) + '.' + String(ip[1]) + '.' + String(ip[2]) + '.' + String(ip[3]);
      s = "HTTP/1.1 200 OK\r\nContent-Type: text/html\r\n\r\n<!DOCTYPE HTML>\r\n<html>Hello from ESP32 at ";
      s += ipStr;
      s += "</html>\r\n\r\n";
      Serial.println("Sending 200");
      client.println(s);  //Send the content of string S. When accessing the E-smart home address using a browser, the information can be read.
  }
  if(req == "/led/on") //Browser accesses IP address/led/on
  {
    client.println("turn on the LED");
    digitalWrite(led_y, HIGH);
  }
  if(req == "/led/off") //Browser accesses IP address/led/off
  {
    client.println("turn off the LED");
    digitalWrite(led_y, LOW);
  }
  if(req == "/fan/on") //Browser accesses IP address/fan/on
  {
    client.println("turn on the fan");
    digitalWrite(fanPin1, LOW); //pwm = 0
    analogWrite(fanPin2, 180);
  }
  if(req == "/fan/off") //Browser accesses IP address/fan/on
  {
    client.println("turn off the fan");
    digitalWrite(fanPin1, LOW); //pwm = 0
    analogWrite(fanPin2, 0);
  }
  //client.print(s);
  client.stop();
}
```

#### **3. Testresultaat**

⚠️ **Opmerking: De mobiele telefoon of tablet moet via dezelfde WiFi met het ESP32 ontwikkelbord verbonden zijn. Anders kan deze de bedieningspagina niet benaderen. Ook verbruikt het ESP32 ontwikkelbord veel stroom wanneer de WiFi-functie wordt gebruikt. Een externe DC-voeding is vereist om in de stroombehoefte voor de werking te voorzien. Als aan de stroombehoefte niet wordt voldaan, zal het ESP32-bord blijven resetten, waardoor de code niet normaal wordt uitgevoerd.**

Als het slimme huis met succes op WiFi is aangesloten, zal het LCD-scherm het toegewezen adres weergeven.
