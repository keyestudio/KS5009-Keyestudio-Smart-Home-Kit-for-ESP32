### 5.4.18 Projekt 10 Tür öffnen


#### **1. Komponentenkenntnisse**

RFID: Der Kartenleser besteht aus einem Funkfrequenzmodul und einem hochfrequenten Magnetfeld. Der Tag-Transponder ist ein passives Sensorgerät ohne Batterie. Er enthält nur kleine integrierte Schaltkreise, ein Medium zur Speicherung von Daten und Antennen zum Empfangen und Senden von Signalen.

Um die Daten im Tag auszulesen, bringen Sie ihn zuerst in die Lesereichweite des Kartenlesers. Der Leser erzeugt ein Magnetfeld, das nach dem Lenzschen Gesetz elektrische Energie induziert; der RFID-Tag versorgt sich dadurch mit Strom und wird aktiviert.

#### **3. Testcode**

```c
//**********************************************************************************
/*
 * Filename    : RFID
 * Description : RFID reader UID
 * Auther      : http//www.keyestudio.com
*/
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C mylcd(0x27,16,2);
#include <ESP32Servo.h>
Servo myservo;
#include <Wire.h>
#include "MFRC522_I2C.h"
// IIC pins default to GPIO21 and GPIO22 of ESP32
// 0x28 is the i2c address of SDA, if doesn't match，please check your address with i2c.
MFRC522 mfrc522(0x28);   // create MFRC522.
#define servoPin  13
#define btnPin 16
boolean btnFlag = 0;

String password = "";

void setup() {
  Serial.begin(115200);           // initialize and PC's serial communication
  mylcd.init();
  mylcd.backlight();
  Wire.begin();                   // initialize I2C
  mfrc522.PCD_Init();             // initialize MFRC522
  ShowReaderDetails();            // dispaly PCD - MFRC522 read carder
  Serial.println(F("Scan PICC to see UID, type, and data blocks..."));

    // Allow allocation of all timers
    ESP32PWM::allocateTimer(0);
    ESP32PWM::allocateTimer(1);
    ESP32PWM::allocateTimer(2);
    ESP32PWM::allocateTimer(3);
    myservo.setPeriodHertz(50);    // standard 50 hz servo
    myservo.attach(servoPin, 1000, 2000); // attaches the servo on pin 18 to the servo object
    // using default min/max of 1000us and 2000us
    // different servos may require different min/max settings
    // for an accurate 0 to 180 sweep

  mylcd.setCursor(0, 0);
  mylcd.print("Card");
}

void loop() {
  //
  if ( ! mfrc522.PICC_IsNewCardPresent() || ! mfrc522.PICC_ReadCardSerial() ) {
    delay(50);
    password = "";
    if(btnFlag == 1)
    {
      boolean btnVal = digitalRead(btnPin);
      if(btnVal == 0)  //If door close button is pressed (active-low)
      {
        Serial.println("close");
        mylcd.setCursor(0, 0);
        mylcd.print("close");
        myservo.write(0);
        btnFlag = 0;
      }
    }
    return;
  }

  // select one of door cards. UID and SAK are mfrc522.uid.

  // save UID
  Serial.print(F("Card UID:"));
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " ");
    //Serial.print(mfrc522.uid.uidByte[i], HEX);
    Serial.print(mfrc522.uid.uidByte[i]);
    password = password + String(mfrc522.uid.uidByte[i]);
  }
  if(password == "")  //Card number is correct,open the door
  {
    Serial.println("open");
    mylcd.setCursor(0, 0);
    mylcd.clear();
    mylcd.print("open");
    myservo.write(180);
    password = "";
    btnFlag = 1;
  }
  else   //Card number error,dispaly error
  {
    password = "";
    mylcd.setCursor(0, 0);
    mylcd.print("error");
  }
  //Serial.println(password);
}

void ShowReaderDetails() {
  //  attain the MFRC522 software
  byte v = mfrc522.PCD_ReadRegister(mfrc522.VersionReg);
  Serial.print(F("MFRC522 Software Version: 0x"));
  Serial.print(v, HEX);
  if (v == 0x91)
    Serial.print(F(" = v1.0"));
  else if (v == 0x92)
    Serial.print(F(" = v2.0"));
  else
    Serial.print(F(" (unknown)"));
  Serial.println("");
  // when returning to 0x00 or 0xFF, may fail to transmit communication signals
  if ((v == 0x00) || (v == 0xFF)) {
    Serial.println(F("WARNING: Communication failure, is the MFRC522 properly connected?"));
  }
}
//**********************************************************************************
```

#### **4. Testergebnis**

Laden Sie den Code hoch, auf dem LCD1602 erscheint "Card". Öffnen Sie den Serial Monitor und stellen Sie die Baudrate auf "115200" ein.

Halten Sie die mitgelieferte Karte in den RFID-Induktionsbereich; auf dem LCD1602 wird "error" angezeigt, und die Ausgabe des Serial Monitors sieht wie in der Abbildung aus:

![Bild54](../media/bb.png)

Geben Sie die "Card UID" aus dem Bild an der im folgenden Screenshot gezeigten Stelle ein (entfernen Sie Leerzeichen in der "Card UID" sowohl im Bild als auch im Serial Monitor; entfernen Sie führende "0" nur, wenn sie vor allen Ziffern steht (z. B. " 0123" → "123"), behalten Sie eine 0, wenn sie einer Zahl folgt (z. B. "601" bleibt "601")):

![Bild55](../media/cc.png)

Laden Sie den Code erneut hoch. Halten Sie die mitgelieferte Karte in den RFID-Induktionsbereich: die Tür dreht sich und öffnet sich, und auf dem LCD1602 wird "open" angezeigt.

Drücken Sie Taste 1, die Tür dreht sich und schließt sich. Wenn Sie jedoch mit einem anderen blauen Induktionsblock über den Leser wischen, zeigt das LCD1602 "Error" an.
